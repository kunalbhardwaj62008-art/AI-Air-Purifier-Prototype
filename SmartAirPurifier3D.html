// Smart Air Purifier — Live Code void loop() { airQuality = analogRead(A0); aqi = map(airQuality, 200, 800, 0, 500); aqi = constrain(aqi, 0, 500); // Relay / Fan control if (aqi > 100) { digitalWrite(RELAY_PIN, HIGH); } // Buzzer alert if (aqi > 150) { for(int i=0; i<3; i++) { tone(BUZZER_PIN, 1000, 200); delay(250); } } updateLED(); updateDisplay(); delay(1000); }
